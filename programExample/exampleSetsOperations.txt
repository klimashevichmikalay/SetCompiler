function Set getSumm(Set set1, Set set2)
{
  Set sum  
  sum = set1 + set2
  
  return sum
}

function showDecartMult(Set set1, Set set2)
{
  Set mult  
  mult = set1 * set2
  print('it is mult:')
  print(mult)
}

function showDiff(Set set1, Set set2)
{
  Set diff  
   print('it is diff:')
  diff = set1 - set2
  print(diff)
}

function showInters(Set set1, Set set2)
{
  Set diff  
   print('it is intersection')
  diff = set1 ~ set2
  print(diff)
}
